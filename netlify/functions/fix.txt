```javascript
    // Handle email form submission - SEND TO BREVO (FIXED VERSION)
    emailForm.addEventListener('submit', async (e) => {
        e.preventDefault();
        
        const name = document.getElementById('dialogName').value;
        const email = document.getElementById('dialogEmail').value;
        
        if (name && email) {
            const submitBtn = emailForm.querySelector('button[type="submit"]');
            const originalText = submitBtn.innerHTML;
            submitBtn.innerHTML = '<i class="fas fa-spinner fa-spin"></i> Subscribing...';
            submitBtn.disabled = true;
            
            try {
                const response = await fetch('/.netlify/functions/add-contact', {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json'
                    },
                    body: JSON.stringify({
                        email: email,
                        name: name,
                        phone: ''
                    })
                });
                
                const result = await response.json();
                
                if (result.success) {
                    userEmail = email;
                    userName = name;
                    
                    // Store in localStorage
                    localStorage.setItem('userName', name);
                    localStorage.setItem('userEmail', email);
                    
                    // Close dialog and show main content
                    emailDialog.classList.remove('active');
                    mainContent.classList.add('visible');
                    
                    // Show success message
                    const welcomeMsg = document.createElement('div');
                    welcomeMsg.style.position = 'fixed';
                    welcomeMsg.style.bottom = '20px';
                    welcomeMsg.style.right = '20px';
                    welcomeMsg.style.background = 'var(--gradient-gold)';
                    welcomeMsg.style.color = 'var(--dark-bg)';
                    welcomeMsg.style.padding = '1rem 1.5rem';
                    welcomeMsg.style.borderRadius = 'var(--border-radius)';
                    welcomeMsg.style.zIndex = '9999';
                    welcomeMsg.style.fontWeight = 'bold';
                    welcomeMsg.style.boxShadow = 'var(--shadow-gold)';
                    welcomeMsg.innerHTML = `Welcome ${name}! You're now subscribed to our updates. 🎉`;
                    document.body.appendChild(welcomeMsg);
                    setTimeout(() => welcomeMsg.remove(), 4000);
                } else {
                    alert('There was an issue subscribing. Please try again or contact us directly.');
                    console.error('API Error:', result.error);
                    submitBtn.innerHTML = originalText;
                    submitBtn.disabled = false;
                    return;
                }
            } catch (error) {
                console.error('Network Error:', error);
                alert('Network error. Please check your connection and try again.');
                submitBtn.innerHTML = originalText;
                submitBtn.disabled = false;
                return;
            }
            
            submitBtn.innerHTML = originalText;
            submitBtn.disabled = false;
        }
    });
```
