// ─── GATE ─────────────────────────────────────
function shouldShowGate() {
  return !localStorage.getItem('gd_gate_skipped') && 
         !localStorage.getItem('gd_email');

function skipGate() {
  localStorage.setItem('gd_gate_skipped', 'true');

  showToast('Welcome! You can always subscribe later from the footer.');
